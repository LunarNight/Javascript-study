##2数组
###2.1.1数组的创建
```
function Print(x){
	console.log(x);	
}

var a = new Array();
//创建一个空的数组
var b = new Array(10);
//创建一个长度为10的空数组
Print(a+"  "+a.length);
Print(b+"  "+b.length);

var c = [2,3,'a','b'];
//数组中类型可以不一致
Print(c);

var d = [6,[2,3],{Q:1,W:2}];
//数组中包含一个数组,一个对象;
Print(d);
```
