###1.2属性的设置
```
var book = new Object();
//创建一个对象,book是一个引用指向新创建的object
book.a=1;
book.b=2;
//以上是  设置属性
book.title = new Object();
book.title.name = 'javascript';
book.title.length = 9;
//嵌套一个对象  book.title指向一个新的object 并为title对象设置属性
console.log(book.a);
console.log(book.title);
console.log(book.title.length);

```

###1.3属性的枚举
```
var circle = {
	x:3,
	y:3,
	radius:2	
}
delete(circle.x);
//删除 x属性
for(var q in circle){
	console.log(q);
}
//遍历属性名称  输出 y,radius  不能获得被删除的属性

console.log(circle.x);
//输出 undefined
```

###1.2构造函数
```
function Rectangle(w,h){
	this.width = w;
	this.height = h;	
//	return 1234;
}
//一般构造函数不需要加返回值   好像加了也没啥问题

var rec = new Rectangle(2,4);
//须使用 new 关键字进行对象的创建

console.log(rec);
console.log(rec.width);
console.log(rec.height);
```

###1.4原型对象和继承
```
function Animal(name){
  this.name = name;
}
Animal.color = "black";
Animal.prototype.say = function(){
  console.log("I'm " + this.name);
};
var cat = new Animal("cat");
 
console.log(
  cat.name, //cat
  cat.height //undefined
);
cat.say(); //I'm cat
 
console.log(
  Animal.name, //Animal
  Animal.color //back
);
Animal.say(); //Animal.say is not a function

```
